✅ Comb Filter Eigenvalue 分析（使用 single-tone sinusoidal）

matlab
% Comb filter 參數
a = 0.7;
tau = 0.1;
Fs = 44100; % 建議與原音檔相同
D = round(tau * Fs);
N = 2048; % 每個正弦波長度

% 頻率掃描（從 0 到 Nyquist）
frequencies = linspace(0, Fs/2, 1000);
gain = zeros(size(frequencies));

% 掃描不同頻率
for k = 1:length(frequencies)
    f = frequencies(k);
    t = (0:N-1)/Fs;
    x = cos(2*pi*f*t); % 單一頻率輸入

    % Comb Filter：y[n] = x[n] + a*y[n-D]
    b = 1;
    a_coeffs = [1, zeros(1,D-1), -a];
    y = filter(b, a_coeffs, x);

    % Gain = output RMS / input RMS
    gain(k) = rms(y)/rms(x);
end

% 畫出頻率響應
figure;
plot(frequencies, gain);
xlabel('Frequency (Hz)');
ylabel('Gain (Output/Input RMS)');
title('Gain of Single-Tone Sinusoidal Input through Comb Reverberator');
grid on;
